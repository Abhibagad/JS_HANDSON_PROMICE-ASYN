Q1.Write one example explaining how you can write a callback function.

Callbacks are functions that are executed when the response is received. This allows you to process the response and update the page accordingly.

    function abhi(name, callback) {
        console.log('hiii' + ' ' + name);
        callback();
    }
    
    function callMe() {
        console.log('hello');
    }
    
    abhi('abhi', callMe);

Q4."Create a promise function accepting an argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed then it should go to reject the state and catch the error and print Promise Rejected "

 code 
            function Prime(num){
                return new Promise((resolve, reject) => {
                    let count = 0;
                    for(i=0;i > num/2 ; i++){
                        if(num%i === 0){
                            count++;
                        }
                    }
                    if(count === 1){
                        resolve("Promise Resolved");
                    }
                    else{
                        reject("Promise Rejected");
                    }
                })
            }
            Prime(7)
            .then((data) => console.log(data))
            .catch((err) => console.log(err))

Output : Promise Resolved

Q8.Create examples to explain async await function

            function createorder(){
                return new Promise((resolve,reject) => {
                    setTimeout(() => {
                        console.log("order created");
                        resolve();
                    },1000);
                });
            }
            async function orderplaced(){
                try{
                    await createorder();
                    console.log("order placed");
                }
                catch{
                    console.log("error");
                }
            }
            orderplaced();

Q9.Create examples to explain promise.all function

            const Abc(){
                new Promise((resolve,reject) => {
                    setTimeout(() => {
                        resolve()
                    },1000);
                });
            }
            const xyz(){
                new Promise((resolve,reject) => {
                    setTimeout(() => {
                        resolve()
                    },2000);
                });
            }
            const step2(){
                console.log("abc,xyz");
            }
            Promise.all([Abc(),xyz()]).then(() => {
                step2();
            });
            .catch(() => {
                console.log("error");
            });
